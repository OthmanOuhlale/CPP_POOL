#include <cctype>

bool ScalarConverter::isFloat(const std::string& literal)
{
    size_t i = 0;
    size_t dot_count = 0;

    if (literal.length() < 4)
        return false;

    if (literal[literal.length() - 1] != 'f')
        return false;

    if (literal[i] == '+' || literal[i] == '-')
        i++;

    while (i < literal.length() - 1)
    {
        if (literal[i] == '.')
            dot_count++;
        else if (!std::isdigit(literal[i]))
            return false;
        i++;
    }

    if (dot_count != 1)
        return false;

    if (literal[0] == '.' || literal[literal.length() - 2] == '.')
        return false;

    return true;
}

#include <cctype>

bool ScalarConverter::isDouble(const std::string& literal)
{
    size_t i = 0;
    size_t dot_count = 0;

    if (literal.length() < 3)
        return false;

    if (literal[i] == '+' || literal[i] == '-')
        i++;

    while (i < literal.length())
    {
        if (literal[i] == '.')
            dot_count++;
        else if (!std::isdigit(literal[i]))
            return false;
        i++;
    }

    if (dot_count != 1)
        return false;

    if (literal[0] == '.' || literal[literal.length() - 1] == '.')
        return false;

    return true;
}
